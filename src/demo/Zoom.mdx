## Zoom

The `useZoom` hook returns a zoom transform that can be used to scale elements.

<Story of={WheelStories.Wheel} />




## Uncontrolled

```tsx
import { useZoom } from 'hyprvis';

function UncontrolledZoom() {
  const ref = useRef();
  const { zoom } = useZoom(ref);

  return (
    <div ref={ref}>
      {/* Content goes here */}
    </div>
  );
}
```

## Controlled

```tsx
import { useZoom } from 'hyprvis';

function UncontrolledZoom() {
  const ref = useRef();
  const [zoom, setZoom] = useState({ k: 1, x: 0, y: 0 });

  useZoom(ref, {
    value: zoom,
    onChange: setZoom
  });

  return (
    <div ref={ref}>
      {/* Content goes here */}
    </div>
  );
}
```
